public function uc() {

$uc_config .= "define('".strtoupper($k)."', '$v');\n";

$html = file_get_contents($filepath);

@file_put_contents($filepath, $html);

if ($r) {
$data = string2array($r['data']);
}
include $this->admin_tpl('system_uc');
}
