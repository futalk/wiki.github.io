public function changeTableVal()
{
if (IS_AJAX_POST) {
$url = null;

}
break;

// 会员中心菜单表
case 'users_menu':
{
Db::name('users_menu')->where('id','gt',0)->update([
'is_userpage' => 0,

]);

}
break;

// 会员投稿功能
case 'archives':
{
if ('arcrank' == $field) {
if (0 == $value) {
$value = -1;

$value = 0;
}
}
}
break;
// 会员产品类型表
case 'users_type_manage':
{

}

// 以下代码可以考虑去掉，与行为里的清除缓存重复 AppEndBehavior.php / clearHtmlCache

/*清除页面缓存*/
